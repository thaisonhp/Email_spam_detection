# cấu trúc thư mục 

```
email_spam_detection/
│
├── data/                  # Thư mục chứa dữ liệu
│   ├── raw/               # Dữ liệu thô chưa xử lý
│   ├── processed/         # Dữ liệu đã xử lý (ví dụ: dữ liệu đã được tiền xử lý)
│   └── sample_emails/     # Ví dụ về các email để thử nghiệm
│
├── notebooks/             # Thư mục chứa các notebook Jupyter (nếu sử dụng)
│   ├── exploratory_data_analysis.ipynb  # Phân tích dữ liệu ban đầu
│   └── model_training.ipynb              # Huấn luyện mô hình
│
├── src/                  # Thư mục chứa mã nguồn
│   ├── __init__.py       # Tập tin khởi tạo cho module
│   ├── data_preprocessing.py  # Mã tiền xử lý dữ liệu
│   ├── feature_extraction.py   # Mã trích xuất đặc trưng (nếu cần)
│   ├── model.py          # Mã định nghĩa mô hình (ví dụ: Naive Bayes)
│   ├── train.py          # Mã huấn luyện mô hình
│   ├── evaluate.py       # Mã đánh giá mô hình
│   └── predict.py        # Mã dự đoán cho email mới
│
├── requirements.txt       # Tập tin liệt kê các thư viện cần thiết
│
├── README.md              # Tập tin mô tả dự án
│
└── main.py                # Tập tin chính để chạy chương trình
